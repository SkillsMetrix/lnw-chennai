import { Directive, ElementRef, OnInit, Renderer2 } from "@angular/core";
import { elementAt } from "rxjs";

@Directive({
    selector: '[appBold]',
    standalone:true
})
export class FontWtDirective implements OnInit {
    private weight = 'bold'
    constructor(private ef: ElementRef, private ren: Renderer2) {

    }
    ngOnInit(): void {
        this.ren.setStyle(
            this.ef.nativeElement, 'font-weight', this.weight
        )
    }
}
